\section{Introducción}

\begin{frame}[c,shrink]{Introducción}
    \begin{block}{Contextualización}
        \begin{itemize}
        	\item Gran cantidad de personas hoy en día intentan cuidar su salud mediante su alimentación pero encuentran repetitivo y costoso utilizar otros sistemas que automatizan esta tarea.
	        \item Para asegurar que se lleva una dieta correcta se deberá hacer un seguimiento de la información nutricional de la comida del usuario.
        	\item Análisis DAFO:
            \begin{itemize}
	            \item Fortalezas: Gran cantidad de consumidores, Facilidad de implementación, Bajo coste.
	            \item Debilidades: Gran cantidad de competición.
	            \item Oportunidades: Sencillez de diseño, información nutricional, solución Open-Source
	            \item Amenazas: Uso costoso para usuario
	        \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\section{Alcance y Objetivos}
\begin{frame}[c]{Objetivos}
    \begin{block}{Objetivos}
    	El objetivo principal es el diseño e implementación de una aplicación web con las siguientes funcionalidades principales:
        \begin{itemize}
        	\item Gestión de ingredientes y platos registrados por el usuario.
	        \item Gestión y seguimiento de menús semanales.
        \end{itemize}
    	Y las siguientes características:
    	\begin{itemize}
	    	\item Diseño sencillo y de facilidad de uso.
	    	\item Interfaz especializada en multiplataforma.
    	\end{itemize}
    \end{block}
\end{frame}

    \subsection{Viabilidad y Planificación}
    \begin{frame}[c]{Viabilidad y Planificación}
        \begin{block}{Análisis de Viabilidad}
            \begin{itemize}
	            \item Encuestas con potenciales usuarios de nuestra aplicación.
	            \item Bajo coste del desarrollo del sistema.
            \end{itemize}
        \end{block}
        \begin{block}{Planificación}
	        El proyecto fue dividido en 15 sprints semanales que veremos en mayor detalle posteriormente.
        \end{block}
    \end{frame}
    
    \subsection{Sprints}
    \begin{frame}[c,shrink]{Sprints}
    \begin{block}{División en sprints}
    	\begin{itemize}
    		\item Sprint 1 - 3: Estudio del arte, requisitos, formación, análisis y diseño.
    		\item Sprint 4 - 6: Gestión básica de ingredientes, platos y menús.
    		\item Sprint 7 - 9: Gestión de usuarios y sus configuraciones. Seguimiento nutricional de menús
    		\item Sprint 10 - 12: Estimación Precio, Recetario, Generaciones de menú y Redes Sociales
    		\item Sprint 13 - 15: UX, Machine Learning, Refactorización y Pruebas exaustivas. 	            
    	\end{itemize}            
    \end{block}
	\end{frame}
    
    \subsection{Tecnología}
    \begin{frame}[c,shrink]{Tecnología}
        \begin{block}{Tecnologías}
            \begin{itemize}
            \item Acceso a datos: MySQL, Hibernate, JDBC, Spring Data
            \item Inyector de dependencias: Spring
            \item Interfaz web: AngularJS
            \end{itemize}
        \end{block}
        \begin{block}{Arquitectura de la Solución}
        	Empleamos dos elementos principales:
        	\begin{itemize}
        		\item Backend EFW-BACK:
        		\begin{itemize}
        			\item Capa Controladora REST
        			\item Capa de Servicios de Negocio
        			\item Capa de acceso a datos.
        		\end{itemize}
        		\item Frontend EFW-FRONT: Componentes reutilizables
        		\begin{itemize}
        			\item Plantilla HTML
        			\item Controlador visual TypeScript
        			\item Servicio Typescript de manejo de datos
        		\end{itemize}
        	\end{itemize}
        \end{block}
    \end{frame}
